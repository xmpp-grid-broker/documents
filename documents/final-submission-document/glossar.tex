\newglossaryentry{broker}{
    name={Broker},
    description={"A \gls{sacm} Broker Controller is a \gls{controller} that contains \gls{control-plane} functions to provide and/or connect services on behalf of other \gls{sacm} components via interfaces on the \gls{control-plane}"~\cite{ietf-sacm-terminology-14}}
}

\newglossaryentry{controller}{
    name={Controller},
    description={"In SACM, a `component containing control plane functions that manage and facilitate information sharing or execute on security functions`; as used here, the term refers to an XMPP server, which provides core message delivery [RFC6120] used by publish-subscribe entities."~\cite{ietf-mile-xmpp-grid-05, ietf-sacm-terminology-14}}
}

\newglossaryentry{control-plane}{
    name={control plane},
    description={"An architectural component that provides common control functions to all \gls{sacm} components."~\cite{ietf-sacm-terminology-14}}
}

\newglossaryentry{component}{
    name={Component},
    description={An encapsulation of software that communicates using
    Interfaces, that is composed of \gls{sacm} capabilities.~\cite{ietf-sacm-terminology-14, ietf-i2nsf-terminology-05}}
}

\newglossaryentry{platform}{
    name={Platform},
    description={"Any entity that connects to the \gls{xmpp}-Grid in order to publish or consume security-related data."~\cite{ietf-mile-xmpp-grid-05}}
}

\newglossaryentry{topic}{
    name={Topic},
    description={"A contextual information channel created on a Broker at which messages generated by a Provider are propagated in real time to one or more Consumers. Each Topic is limited to a specific type and format of security data (e.g., IODEF) and provides an XMPP interface by which the data can be obtained."~\cite{ietf-mile-xmpp-grid-05}}
}

\newglossaryentry{provider}{
    name={Provider},
    description={"In \gls{sacm} An entity that contains functions to provide information to other \glspl{component}; as used here, the term refers to an \gls{xmpp} \gls{publish-subscribe} Publisher."~\cite{ietf-mile-xmpp-grid-05}}
}

\newglossaryentry{consumer}{
    name={Consumer},
    description={"In \gls{sacm}, an entity that contains functions to receive information from other \glspl{component}; as used here, the term refers to an \gls{xmpp} \gls{publish-subscribe} Subscriber."~\cite{ietf-mile-xmpp-grid-05}}
}
